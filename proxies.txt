# مثال على ملف البروكسي
# ضع بروكسي واحد في كل سطر
# يمكن استخدام تنسيقات مختلفة:
# ip:port
# http://ip:port
# http://username:password@ip:port
# socks5://ip:port

# أمثلة (هذه مجرد أمثلة وهمية):
# 192.168.1.1:8080
# http://proxy1.example.com:3128
# http://user:pass@proxy2.example.com:8080
# socks5://127.0.0.1:1080
